      Relation between temperature and speed of sound

      import pandas as pd
      import numpy as np
      from matplotlib import pyplot as plt
      # Importing data from .csv file and build numpy matrix
      csv = pd.read_csv("./data0.csv", sep=",")
      data = csv.to_numpy()
      data = np.transpose(data)
      # Moving average function
      def m_a(x, w):
          return np.convolve(x, np.ones(w), 'valid') / w
      # Averaging data
      tmp = m_a(data[0],30)
      hum = m_a(data[1],30)
      time = m_a(data[2],30)
      # Calculate speed of sound
      def speed(t):
          c = np.zeros(t.shape[0])
          for i in range(t.shape[0]):
              c[i] = (2*0.5)/(t[i]*10**(-6))
          return c
      cS = speed(time)
      # Linear regression
      linear_f = np.poly1d(np.polyfit(tmp,cS,1))
